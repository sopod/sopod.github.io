struct Plane
{
public:
    FORCEINLINE constexpr float Distance(const Vector3& InPoint) const;
    FORCEINLINE constexpr bool IsOutside(const Vector3& InPoint) const;
    
    Vector3 Normal = Vector3::UnitY;  // (a, b, c)
    float D = 0.f;  // ÏÉÅÏàò dÏùò Í∞í
};

Plane::Plane(const Vector4& InVector4)
{
    Normal = InVector4.ToVector3();  // (a, b, c)
    D = InVector4.W;  // ÏÉÅÏàò dÏùò Í∞í
    Normalize();
}

// Ï†ïÍ∑úÌôî ÌïúÎã§. 
void Plane::Normalize()
{
    // a^2 + b^2 + c^2
    float squaredSize = Normal.SizeSquared();
    
    // ÌÅ¨Í∏∞Í∞Ä 1Ïù¥Î©¥ Ï†ïÍ∑úÌôîÌï† ÌïÑÏöî ÏóÜÎã§. 
    if (Math::EqualsInTolerance(squaredSize, 1.f))
    {
        return;
    }
    
    // 1 / sqrt(a^2 + b^2 + c^2)ÏùÑ Í≥†ÏÜç Ïó≠Ï†úÍ≥±Í∑º(Fast inverse square root) Ìï®ÏàòÎ°ú Íµ¨ÌïúÎã§. 
    float invLength = Math::InvSqrt(squaredSize);
    
    // Í≥±Ìï¥ÏÑú Ï†ïÍ∑úÌôîÌïúÎã§. 
    Normal *= invLength;
    D *= invLength;
}


// p + dÎ•º Î¶¨ÌÑ¥ÌïúÎã§. 
FORCEINLINE constexpr float Plane::Distance(const Vector3& InPoint) const
{
    return Normal.Dot(InPoint) + D;
}

// DistanceÏùò Î∞òÌôòÍ∞íÏù¥ ÏñëÏàòÏù¥Î©¥ Í∑∏ Ï†êÏùÄ ÌèâÎ©¥Ïùò Î∞îÍπ•Ïóê ÏûàÎã§. 
FORCEINLINE constexpr bool Plane::IsOutside(const Vector3& InPoint) const
{
    return Distance(InPoint) > 0.f;
}

// Î†åÎçîÎßÅ Î°úÏßÅÏùÑ Îã¥ÎãπÌïòÎäî Ìï®Ïàò
void SoftRenderer::Render3D()
{
    // Î†åÎçîÎßÅ Î°úÏßÅÏóêÏÑú ÏÇ¨Ïö©ÌïòÎäî Î™®Îìà ÎÇ¥ Ï£ºÏöî Î†àÌçºÎü∞Ïä§
    const GameEngine& g = Get3DGameEngine();
    auto& r = GetRenderer();
    const CameraObject& mainCamera = g.GetMainCamera();

    // Î∞∞Í≤ΩÏóê Í∏∞Ï¶àÎ™® Í∑∏Î¶¨Í∏∞
    DrawGizmo3D();

    // Î†åÎçîÎßÅ Î°úÏßÅÏùò Î°úÏª¨ Î≥ÄÏàò
    const Matrix4x4 vMatrix = mainCamera.GetViewMatrix();
    const Matrix4x4 pMatrix = mainCamera.GetPerspectiveMatrix();  // ÏõêÍ∑º Ìà¨ÏòÅ ÌñâÎ†¨ÏùÑ Í∞ÄÏ†∏Ïò®Îã§. 
    const Matrix4x4 pvMatrix = mainCamera.GetPerspectiveViewMatrix();

    // Ï†àÎëêÏ≤¥ Íµ¨Ï∂ïÏùÑ ÏúÑÌïú Ìà¨ÏòÅ ÌñâÎ†¨Ïùò ÏÑ§Ï†ï
    Matrix4x4 ptMatrix = pMatrix.Transpose();

    // Ï†àÎëêÏ≤¥Î•º Íµ¨ÏÑ±ÌïòÎäî ÌèâÎ©¥Ïùò Î∞©Ï†ïÏãù
    // Ï†àÎëêÏ≤¥Î•º Íµ¨ÏÑ±ÌïòÎäî 6Í∞úÏùò ÌèâÎ©¥ÏùÑ ÏÉùÏÑ±ÌïúÎã§. 
    std::array<Plane, 6> frustumPlanes = {
        Plane(-(ptMatrix[3] - ptMatrix[1])), // +Y
        Plane(-(ptMatrix[3] + ptMatrix[1])), // -Y
        Plane(-(ptMatrix[3] - ptMatrix[0])), // +X
        Plane(-(ptMatrix[3] + ptMatrix[0])), // -X
        Plane(-(ptMatrix[3] - ptMatrix[2])), // +Z
        Plane(-(ptMatrix[3] + ptMatrix[2])), // -Z
    };

    // Ï†àÎëêÏ≤¥ ÏÑ†Ïñ∏ 
    Frustum frustumFromMatrix(frustumPlanes); // ÌèâÎ©¥Ïùò Î∞∞Ïó¥ÏùÑ Ïù∏ÏûêÎ°ú ÎÑ£Ïñ¥ Ï†àÎëêÏ≤¥Î•º Íµ¨Ï∂ïÌïúÎã§. 

    // Ï†àÎëêÏ≤¥ Ïª¨ÎßÅ ÌÖåÏä§Ìä∏Î•º ÏúÑÌïú ÌÜµÍ≥Ñ Î≥ÄÏàò 
    size_t totalObjects = g.GetScene().size(); // Ïî¨Ïóê ÏÜçÌïú Ï†ÑÏ≤¥ Í≤åÏûÑ Ïò§Î∏åÏ†ùÌä∏ ÏàòÎ•º ÌååÏïÖÌïúÎã§. 
    size_t culledObjects = 0; // Î†åÎçîÎßÅÏù¥ ÏßÑÌñâÎêú Í≤åÏûÑ Ïò§Î∏åÏ†ùÌä∏ ÏàòÎ•º ÌååÏïÖÌïúÎã§. 
    size_t renderedObjects = 0; // Î†åÎçîÎßÅÏù¥ ÏßÑÌñâÎêú Í≤åÏûÑ Ïò§Î∏åÏ†ùÌä∏ ÏàòÎ•º ÌååÏïÖÌïúÎã§. 

    for (auto it = g.SceneBegin(); it != g.SceneEnd(); ++it)
    {
        const GameObject& gameObject = *(*it);
        if (!gameObject.HasMesh() || !gameObject.IsVisible())
        {
            continue;
        }

        // Î†åÎçîÎßÅÏóê ÌïÑÏöîÌïú Í≤åÏûÑ Ïò§Î∏åÏ†ùÌä∏Ïùò Ï£ºÏöî Î†àÌçºÎü∞Ïä§Î•º ÏñªÍ∏∞
        const Mesh& mesh = g.GetMesh(gameObject.GetMeshKey());
        const TransformComponent& transform = gameObject.GetTransform();

        // Ï†àÎëêÏ≤¥ Ïª¨ÎßÅ Íµ¨ÌòÑ
        // Í≤åÏûÑ Ïò§Î∏åÏ†ùÌä∏Ïùò ÏúÑÏπòÎ•º Î∑∞ Í≥µÍ∞ÑÏúºÎ°ú Î≥ÄÌôòÌïúÎã§. 
        Vector4 viewPos = vMatrix * Vector4(transform.GetPosition());
        
        // Î∑∞ Í≥µÍ∞ÑÏùò ÏúÑÏπòÍ∞Ä Ï†àÎëêÏ≤¥ ÏòÅÏó≠Ïùò Ïô∏Î∂ÄÏóê ÏûàÎäîÏßÄ Í≤ÄÏÇ¨ÌïúÎã§. 
        if (frustumFromMatrix.CheckBound(viewPos.ToVector3()) == BoundCheckResult::Outside)
        {
            // Í∑∏Î¶¨ÏßÄ ÏïäÍ≥† Í±¥ÎÑàÎúÄ
            culledObjects++;
            continue;
        }

        // ÏµúÏ¢Ö ÌñâÎ†¨ Í≥ÑÏÇ∞
        Matrix4x4 finalMatrix = pvMatrix * transform.GetModelingMatrix();

        // Î©îÏãú Í∑∏Î¶¨Í∏∞
        DrawMesh3D(mesh, finalMatrix, gameObject.GetColor());

        // Í∑∏Î¶∞ Î¨ºÏ≤¥Î•º ÌÜµÍ≥ÑÏóê Ìè¨Ìï®
        renderedObjects++;
    }

    r.PushStatisticText("Total GameObjects : " + std::to_string(totalObjects));
    r.PushStatisticText("Culled GameObjects : " + std::to_string(culledObjects));
    r.PushStatisticText("Rendered GameObjects : " + std::to_string(renderedObjects));
}
